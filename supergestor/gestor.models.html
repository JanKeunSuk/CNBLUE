    def create_user(self, username, email, password=None):
        """
        Crea y guarda un usuario con el nombre de usuario , email y contrasea
        dados
        """
        if not username:
            raise ValueError('Los usuarios deben tener un nombre de usuario')

        user = self.model(
            username=username,
            email=Permitido.objects.get(pk=email),
        )
        user.set_password(password)
        user.save(using=self._db)
        return user

    def create_superuser(self, username, email, password):
        """
        Crea y guarda un superuser con el nombre de usuario, email y password
        dados
        """
        user = self.create_user(username,
            password=password,
            email=Permitido.objects.get(pk=email),
        )
        user.is_admin =True
        user.save(using=self._db)
        return user
